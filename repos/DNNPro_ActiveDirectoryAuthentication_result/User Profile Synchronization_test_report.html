private void UpdateDNNUser(UserInfo objReturnUser, ADUserInfo objAuthUser)
{
    if (!(objAuthUser.DisplayName == ""))
    {
        objReturnUser.DisplayName = objAuthUser.DisplayName;
    }

    if (!(objAuthUser.Email == ""))
    {
        objReturnUser.Email = objAuthUser.Email;
    }
    // ... (similar checks for all other fields)

    UserController objAuthUserController = new UserController();
    objAuthUserController.UpdateDnnUser(objReturnUser);
}

HTTP Error 500.19 - IIS Windows Authentication configuration issue

Issue Found

Error: HTTP Error 500.19 - Internal Server Error

Config Error: "This configuration section cannot be used at this path. This happens when the section is locked at a parent level."

Affected Settings:
  • windowsAuthentication enabled="true"
  • anonymousAuthentication enabled="false"

Resolution: The IIS Windows Authentication module settings need to be unlocked at the server level using IIS Manager or by running:
%windir%\system32\inetsrv\appcmd unlock config -section:system.webServer/security/authentication/windowsAuthentication
